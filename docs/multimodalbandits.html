  75def generate_multimodal_function(G,m,mmax,sigma):
  76    """ Generate a multimodal function over a graph with a specfied set of modes using a gaussian mixture function. Warning: if sigma is too large, the gaussian peaks might "merge" so that the generated function is less than len(m)-multimodal
  77    Args:
  78        G (networkx graph): The graph G
  79        m (list): The set of modes 
  80        mmax (float): The index of the largest mode
  81        sigma (float): The standard deviation for each gaussian mixture component
  82
  83    Returns:
  84        mu (numpy array):  A multimodal function over graph G with set of modes equal to m
  85
  86    """
  87    mu = np.zeros(G.number_of_nodes())
  88    d = dict(nx.all_pairs_shortest_path_length(G))
  89    for j in m:
  90        for i,_ in enumerate(mu):
  91            mu[i] = mu[i] + (1 + (j == mmax))*np.exp(-(1/(sigma))*d[i][j])
  92    return(mu)

  94def generate_spread_out_modes(nb_arms, nb_modes):
  95    """ For a line graph with nodes {0,...,nb_arms-1} and nb_modes modes, computes a set of spread out modes that maximizes the number of points outside of their neighborhood. Helpful to get a more representative runtime in the runtime experiment
  96    Args:
  97        nb_arms (int): The number of arms
  98        nb_modes (int): The number of modes
  99
 100    Returns:
 101        modes: The set of modes
 102
 103    """
 104    if nb_modes * 3 - 1 > nb_arms:
 105        raise ValueError(f"Cannot generate {nb_modes} spread out modes for {nb_arms} arms.")
 106    
 107    modes = []
 108    available_positions = list(range(nb_arms))
 109    # Start with the extremes
 110    modes.append(0)
 111    modes.append(nb_arms - 1)
 112    available_positions = available_positions[2:-2]  # Remove the extremes and their neighbors
 113    
 114    # Add remaining modes
 115    for _ in range(nb_modes - 2):
 116        if not available_positions:
 117            raise ValueError("Not enough positions to place modes.")
 118        
 119        # Choose the middle of the largest gap
 120        gaps = [available_positions[i+1] - available_positions[i] for i in range(len(available_positions)-1)]
 121        largest_gap_index = gaps.index(max(gaps))
 122        new_mode = (available_positions[largest_gap_index] + available_positions[largest_gap_index+1]) // 2
 123        
 124        modes.append(new_mode)
 125        
 126        # Remove the new mode and its neighbors from available positions
 127        available_positions = [p for p in available_positions if abs(p - new_mode) > 1]
 128    
 129    return sorted(modes)
 130def get_random_modes(G, nb_modes, k_star):
 131    """For a graph G, computes a set of nb_modes random modes that contain the optimal arm k_star
 132
 133    Args:
 134        G (networkx graph): The graph G 
 135        nb_modes (int): The number of modes
 136        k_star (int): The optimal arm
 137
 138    Returns:
 139        modes: The set of modes
 140
 141    """
 142    modes = {k_star}
 143    forbidden = {k_star} | set(G.neighbors(k_star))
 144    available_nodes = [n for n in G.nodes() if n not in forbidden]
 145    
 146    while len(modes) < nb_modes and available_nodes:
 147        new_mode = np.random.choice(available_nodes)
 148        modes.add(new_mode)
 149        forbidden.update({new_mode} | set(G.neighbors(new_mode)))
 150        available_nodes = [n for n in available_nodes if n not in forbidden]
 151        
 152    return list(modes)

 171def regression_graph(G,mu,eta,p,k,N):
 172    """Compute the minimizer of the weighted sum of divergences sum_k eta_k d(mu_k,lambda_k) over all multimodal functions lambda with a fixed set of modes, discretized over a grid
 173
 174    Args:
 175        G (networkx graph): The graph G
 176        mu (numpy array): The values of function mu
 177        eta (numpy array): The values of weights eta
 178        p (list): The location of the modes of lambda
 179        k (int): The maximal entry of lambda
 180        N (int): The number of grid points used for discretization
 181
 182    Returns:
 183        lambdastar (numpy array): The values of minimizer lambda
 184
 185    """
 186    INF = 1e10
 187    K = mu.shape[0]
 188    kstar = int(np.argmax(mu))
 189    grid = np.linspace(np.min(mu), np.max(mu), N)
 190    T = nx.bfs_tree(G, k)
 191
 192    # Allocate arrays
 193    f = np.zeros([K, N, 2])              # f[...,0] is u=-1 (<=), f[...,1] is u=+1 (>)
 194    fstar = np.zeros([K, N, 2])         # fstar[...,0] = min_{j<=i} f[...,0], fstar[...,1] = min_{j>i} f[...,1]
 195    fsquare = np.zeros([K, N])          # f^\diamond = min(fstar[...,0], fstar[...,1])
 196    v = np.full([K, N], -1, dtype=int)  # which child attains min_{child} g_child(z)
 197    lambdastar = np.zeros(K, dtype=float)
 198
 199    # Pointers for backtracking 
 200    ptr_minus = np.full([K, N], -1, dtype=int)  # argmin index in [0..i] for f[...,0]
 201    ptr_plus  = np.full([K, N], -1, dtype=int)  # argmin index in [i+1..N-1] for f[...,1], -1 if none
 202
 203    # Forward DP: compute f, then fstar/ptr arrays and fsquare
 204    for ell in reversed(list(T.nodes())):
 205        children = sorted(list(T.successors(ell)))  # deterministic order, just in case
 206        # effective eta: make k^\star have infinite weight so its lambda must equal mu^\star
 207        e = INF if (ell == kstar) else float(eta[ell])
 208
 209        # compute f[ell,i,u] for all grid points
 210        for i, z in enumerate(grid):
 211            div = e * divergence(mu[ell], z)
 212            if ell in p:  # ell is allowed to be a mode
 213                base = div
 214                # sum over children of their fsquare at same z
 215                s = 0.0
 216                for j in children:
 217                    s += fsquare[j, i]
 218                f[ell, i, 0] = base + s
 219                f[ell, i, 1] = base + s
 220            else:
 221                # ell cannot be a mode, and needs to have a neighbour whose reward is higher or equal
 222                s = 0.0
 223                for j in children:
 224                    s += fsquare[j, i]
 225                f[ell, i, 0] = div + s
 226                # compute the extra min_v term for u=+1 if children exist
 227                if children:
 228                    # for each child compute g_child = min(fstar_child(+1), f_child(-1)) - fsquare_child
 229                    min_g_cost = INF
 230                    winner = -1
 231                    for child in children:
 232                        #  f[child, i, 0] is f_child(z,-1)
 233                        cand_plus = fstar[child, i, 1]   # min_{w>z} f_child(w,+1)
 234                        cand_minus = f[child, i, 0]      # f_child(z, -1)
 235                        g_child = min(cand_plus, cand_minus) - fsquare[child, i]
 236                        if g_child < min_g_cost:
 237                            min_g_cost = g_child
 238                            winner = child
 239                    v[ell, i] = int(winner)  # store the winning child id
 240                    f[ell, i, 1] = div + s + min_g_cost
 241                else:
 242                    f[ell, i, 1] = INF
 243
 244        # Compute fstar and pointers for ell:
 245        # fstar[...,0] = min_{j<=i} f[...,0]  and ptr_minus is the argmin in [0..i]
 246        fstar[ell, 0, 0] = f[ell, 0, 0]
 247        ptr_minus[ell, 0] = 0
 248        for i in range(1, N):
 249            if f[ell, i, 0] <= fstar[ell, i-1, 0]:
 250                fstar[ell, i, 0] = f[ell, i, 0]
 251                ptr_minus[ell, i] = i
 252            else:
 253                fstar[ell, i, 0] = fstar[ell, i-1, 0]
 254                ptr_minus[ell, i] = ptr_minus[ell, i-1]
 255
 256        # fstar[...,1] = min_{j>i} f[...,1] ; for i = N-1 it is INF and ptr_plus = -1
 257        fstar[ell, N-1, 1] = INF
 258        ptr_plus[ell, N-1] = -1
 259        # iterate backwards to compute min over j > i
 260        for i in range(N - 2, -1, -1):
 261            # compare immediate next f at i+1 vs min at i+1
 262            if f[ell, i+1, 1] <= fstar[ell, i+1, 1]:
 263                fstar[ell, i, 1] = f[ell, i+1, 1]
 264                ptr_plus[ell, i] = i + 1
 265            else:
 266                fstar[ell, i, 1] = fstar[ell, i+1, 1]
 267                ptr_plus[ell, i] = ptr_plus[ell, i+1]
 268
 269        #  compute fsquare
 270        for i in range(N):
 271            fsquare[ell, i] = min(fstar[ell, i, 0], fstar[ell, i, 1])
 272
 273    # Backward pass using pointers
 274    
 275    lambdastar[k] = np.max(mu)
 276    # iterate nodes in BFS order from root (k)
 277    for ell in list(T.nodes()):
 278        if ell == k:
 279            continue
 280
 281        parent = next(T.predecessors(ell))
 282        # robust nearest-grid index for parent's lambda (avoid float == just in case) 
 283        parent_grid_index = int(np.argmin(np.abs(grid - lambdastar[parent])))
 284
 285        # Check if ell is constrained child
 286        grandparent_list = list(T.predecessors(parent))
 287        ell_is_constrained = False
 288        if grandparent_list:
 289            grandparent = grandparent_list[0]
 290            if lambdastar[parent] > lambdastar[grandparent] and parent not in p:
 291                constrained_child = v[parent, parent_grid_index]
 292                if constrained_child == ell:
 293                    ell_is_constrained = True
 294
 295
 296
 297        
 298        # Decide value of lambda_ell using stored pointers
 299        
 300        if ell_is_constrained:
 301            if fstar[ell, parent_grid_index, 1] <= f[ell, parent_grid_index, 0]:
 302                j = ptr_plus[ell, parent_grid_index]  # index j > parent_grid_index minimizing f[...,1]
 303                if j == -1:
 304                    # no strictly greater grid value -> fallback to equality with parent
 305                    lambdastar[ell] = lambdastar[parent]
 306                else:
 307                    lambdastar[ell] = grid[j]
 308            else:
 309                lambdastar[ell] = lambdastar[parent]
 310        else:
 311            # unconstrained case: compare fstar(+1) vs fstar(-1)
 312            if fstar[ell, parent_grid_index, 1] <= fstar[ell, parent_grid_index, 0]:
 313                j = ptr_plus[ell, parent_grid_index]
 314                if j == -1:
 315                    lambdastar[ell] = lambdastar[parent]
 316                else:
 317                    lambdastar[ell] = grid[j]
 318            else:
 319                j = ptr_minus[ell, parent_grid_index]
 320                if j == -1:
 321                    lambdastar[ell] = grid[0]
 322                else:
 323                    lambdastar[ell] = grid[j]
 324
 325    return lambdastar

 333def regression_all(G,mu,eta,N,nb_modes):
 334    """Compute the minimizer of the weighted sum of divergences sum_k eta_k d(mu_k,lambda_k) over all multimodal functions lambda with a fixed number of modes, discretized over a grid
 335
 336    Args:
 337        G (networkx graph): The graph G
 338        mu (numpy array): The values of function mu
 339        eta (numpy array): The values of weights eta
 340        N (int): The number of grid points used for discretization
 341        nb_modes (int): The number of modes of lambda 
 342
 343    Returns:
 344        (lambdastar,vstar) (numpy array, float): The values of minimizer lambda and the minimal value of the weighted sum of divergences 
 345
 346    """
 347    # Computes explicitly the solution of PGL(k) when k is in the neighborhood of a mode or mu has strictly less than m modes, performs dynamic programming for the other k's
 348    kstar = np.argmax(mu)
 349    m = compute_modes(G,mu)
 350    K=mu.shape[0]
 351    lambdastar = np.ones(K)*np.max(mu)
 352    vstar = sum(eta*divergence(mu,lambdastar)) 
 353    if nb_modes > len(m): #if mu is strictly less than m-modal, we have no constraints besides lambda[k]=lambda[kstar] for k different than kstar
 354        for k in [k_val for k_val in range(K) if k_val != kstar]:
 355            lambdastar_new=np.copy(mu)
 356            lambdastar_new[k]=mu[kstar]
 357            if (vstar > sum(eta*divergence(mu,lambdastar_new))):
 358                lambdastar = lambdastar_new
 359                vstar = sum(eta*divergence(mu,lambdastar))
 360    else:
 361        neighborhood=modes_neighborhood(G,mu)
 362        if kstar in neighborhood:
 363            neighborhood.remove(kstar)
 364            remote_arms = [k for k in range(K) if k != kstar and k not in neighborhood]
 365        for k in neighborhood:
 366            lambdastar_new=np.copy(mu)
 367            lambdastar_new[k]=mu[kstar]
 368            if (vstar > sum(eta*divergence(mu,lambdastar_new))):
 369                lambdastar = lambdastar_new
 370                vstar = sum(eta*divergence(mu,lambdastar))
 371                
 372        if nb_modes == 1:
 373            return (lambdastar, vstar) # This is the optimal solution in the unimodal case
 374     
 375        remote_arms = [k for k in range(K) if k != kstar and k not in neighborhood]
 376        for k in remote_arms:
 377            for j in m:
 378                p = list(m);p.remove(j);p.append(k)
 379                lambda_new=np.copy(mu)
 380                lambda_new[k]=mu[kstar]
 381                if vstar>sum(eta*divergence(mu,lambda_new)): # If this is not the case, it is unecessary to go further
 382                    lambdastar_new = regression_graph(G,mu,eta,p,k,N)
 383                    if (vstar > sum(eta*divergence(mu,lambdastar_new))):
 384                        lambdastar = lambdastar_new
 385                        vstar = sum(eta*divergence(mu,lambdastar))
 386    return (lambdastar,vstar)

 391def fast_minimization(g,b,c):
 392    """ Compute the minimization of sum_{v in C(ell)} g(b_v,c_v) under constraints that sum_{v in C(ell)} b_v = b and sum_{v in C(ell)} c_v = c  using "fast minimization"
 393
 394    Args:
 395        g (numpy array): The values of g, with size (size(C) x 2 x 2) 
 396        b (int): The value of sum_{v in C(ell)} b_v. Should be either 0 or 1.
 397        c (int): The value of sum_{v in C(ell)} c_v. Should be either 0 or 1.
 398
 399    Returns:
 400        (vstar,bstar,cstar) (float,numpy array,numpy array): the minimal value as well as the optimal solution
 401
 402    """
 403    vstar = 10**10
 404    bstar = [0 for i in range( g.shape[0] )] 
 405    cstar = [0 for i in range( g.shape[0] )] 
 406    if (b==0) and (c==0):
 407        vstar = np.sum(g[:,0,0])
 408    elif (b==1) and (c==0): 
 409        vstar = np.min(  g[:,1,0] - g[:,0,0] ) + np.sum(g[:,0,0])
 410        bstar[np.argmin(  g[:,1,0] - g[:,0,0] ) ] = 1
 411    elif (b==0) and (c==1): 
 412        vstar = np.min(  g[:,0,1] - g[:,0,0] )  + np.sum(g[:,0,0])
 413        cstar[np.argmin(  g[:,0,1] - g[:,0,0] ) ] = 1
 414    elif (b==1) and (c==1): 
 415        # first case, b_v = c_v for some node v 
 416        v11 = np.argmin(g[:,1,1] - g[:,0,0])
 417        D11 = g[v11,1,1] - g[v11,0,0]
 418        #second case, there exists two distinct nodes w_1,w_2 such that b_{w_1} = 1 = c_{w_2}, b_{w_2} = 0 = c_{w_1}
 419        # in this case one must be careful if both A01 and A10 are minimized at the same entry, if so, we must chose the first best and second best 
 420        # note: this case is only possible if there are at least two entries
 421        if (g.shape[0] > 1):
 422            A10 = g[:,1,0] - g[:,0,0]
 423            v10 = A10.argsort()
 424            A01 = g[:,0,1] - g[:,0,0]
 425            v01 = A01.argsort()
 426            if (v10[0] == v01[0]):
 427                D1001 = min(  A10[v10[0]] + A01[v01[1]] ,  A10[v10[1]] + A01[v01[0]])
 428            else:
 429                D1001 = A10[v10[0]] + A01[v01[0]]
 430            vstar = min(D11,D1001)+ np.sum(g[:,0,0])
 431            if D11 < D1001: 
 432                bstar[v11] = 1
 433                cstar[v11] = 1
 434            else:
 435                if (v01[0] == v10[0]):
 436                    if (A10[v10[0]] + A01[v01[1]] <  A10[v10[1]] + A01[v01[0]]):
 437                        bstar[v10[0]] = 1
 438                        cstar[v01[1]] = 1
 439                    else:
 440                        bstar[v10[1]] = 1
 441                        cstar[v01[0]] = 1
 442                else:
 443                        bstar[v10[0]] = 1
 444                        cstar[v01[0]] = 1
 445        else:
 446            bstar[0] = 1
 447            cstar[0] = 1
 448            vstar = D11+ np.sum(g[:,0,0])
 449    return(vstar,bstar,cstar)

 452def fast_dynamic_programming(G,mu,eta,N,nb_modes):
 453    """Compute the minimizer of the weighted sum of divergences sum_k eta_k d(mu_k,lambda_k) over all multimodal functions lambda with a fixed number of modes, discretized over a grid, using a more elaborate method than regression_all. This method is expected to be faster for large graphs of high degree.
 454
 455    Args:
 456        G (networkx graph): The graph G
 457        mu (numpy array): The values of function mu
 458        eta (numpy array): The values of weights eta
 459        N (int): The number of grid points used for discretization
 460        nb_modes (int): The number of modes of lambda 
 461
 462    Returns:
 463        (lambdastar,vstar) (numpy array, float): The values of minimizer lambda and the minimal value of the weighted sum of divergences 
 464
 465    """
 466    # Find the minimal value of the weighted divergence with m modes, discretization number N
 467    M = compute_modes(G,mu)
 468    K = mu.shape[0]
 469    kstar = np.argmax(mu)
 470
 471   # Case number 1: when the maximal entry of confusing parameter lambda does lie in the neighbourhood of a mode of mu, or when the number of modes of lambda is strictly greater than that of mu
 472   # Computes explicitly the solution of PGL(k) when k is in the neighborhood of a mode or mu has strictly less than m modes, performs dynamic programming for the other k's
 473    lambdastar1 = np.ones(K)*np.max(mu)
 474    vstar1 = sum(eta*divergence(mu,lambdastar1)) 
 475    if nb_modes > len(M): # if mu is strictly less than m-modal, we have no constraints besides lambda[k]=lambda[kstar] for k different than kstar
 476        for k in [k_val for k_val in range(K) if k_val != kstar]:
 477            lambdastar_new = np.copy(mu)
 478            lambdastar_new[k]= mu[kstar]
 479            if (vstar1 > sum(eta*divergence(mu,lambdastar_new))):
 480                lambdastar1 = lambdastar_new
 481                vstar1 = sum(eta*divergence(mu,lambdastar1))
 482        return (lambdastar1,vstar1) 
 483    else:
 484        neighborhood = modes_neighborhood(G,mu)
 485        for k in [k_val for k_val in neighborhood if k_val != kstar]:
 486            lambdastar_new = np.copy(mu)
 487            lambdastar_new[k] = mu[kstar]
 488            if (vstar1 > sum(eta*divergence(mu,lambdastar_new))):
 489                lambdastar1 = lambdastar_new
 490                vstar1 = sum(eta*divergence(mu,lambdastar1))
 491        if nb_modes == 1:
 492            return (lambdastar1,vstar1) # This is the best solution in the unimodal setting
 493
 494    # Case number 2: when the maximal entry of confusing parameter lambda does not lie in the neighbourhood of a mode of mu
 495    # Discretize the space of lambda with a grid of size N
 496    grid = np.linspace(np.min(mu),np.max(mu),N)
 497    T = nx.bfs_tree(G, kstar)
 498
 499    # Initialize value arrays
 500    h = np.zeros([K,len(grid),3,2,2])
 501    hplus = np.zeros([K,len(grid),2,2])
 502    hminus = np.zeros([K,len(grid),2,2])
 503    hequal = np.zeros([K,len(grid),2,2])
 504    hstar = np.zeros([K,len(grid),2,2])
 505    hplusequal = np.zeros([K,len(grid),2,2])
 506    
 507    # Pointers to store the index that minimizes hminus, hplus
 508    # Shape: (K, N, b=2, c=2)
 509    ptr_hminus = np.zeros([K, len(grid), 2, 2], dtype=int)
 510    ptr_hplus = np.zeros([K, len(grid), 2, 2], dtype=int)
 511
 512 
 513    max_deg = max((len(list(T[ell])) for ell in T.nodes()), default=0)
 514    # Pointers for the a==1 case
 515    # Shape: (K, N, b=2, c=2, max_deg)
 516    best_bs_a1 = np.zeros([K, len(grid), 2, 2, max_deg], dtype=int)
 517    best_cs_a1 = np.zeros([K, len(grid), 2, 2, max_deg], dtype=int)
 518    best_w_idx_a1 = np.full([K, len(grid), 2, 2], -1, dtype=int)
 519    candidate_w_idx_a1 = np.full([K, len(grid), 2, 2], -1, dtype=int)
 520
 521    # Forward pass over the nodes, sorted by decreasing depth
 522    for ell in reversed(list(T.nodes())):
 523        children = list(T[ell])
 524        num_children = len(children)
 525
 526        if not(children):  # leaf
 527            for a in range(3):
 528                for b in range(2):
 529                    for c in range(2):
 530                        is_max = ((a == 0) and (b==1) and (ell not in M))
 531                        valid = (a != 1) and (b == (a==0)*(ell not in M)) and (c == (a!=0)*(ell in M))
 532                        for i,z in enumerate(grid):
 533                            if (not valid) or (is_max and (i != N-1)):
 534                                h[ ell,i,a,b,c ] = 1e10
 535                            else:
 536                                h[ ell,i,a,b,c ] = eta[ell]*divergence(mu[ell],z)
 537                                
 538        else:  # Internal node
 539            # For each grid point i and each of the 4 child-flag configurations,
 540            # find the candidate child  that offers the minimum penalty
 541            if num_children > 0:
 542                for i in range(N):
 543                    for b_child in range(2):
 544                        for c_child in range(2):
 545                            # Calculate penalties h_ge - h_star for all children at once.
 546                            penalties = hplusequal[children, i, b_child, c_child] - hstar[children, i, b_child, c_child]
 547                            candidate_w_idx_a1[ell, i, b_child, c_child] = np.argmin(penalties)
 548
 549            for i,z in enumerate(grid):
 550                # Construct the candidate set W_ell(z)
 551                candidate_indices = {
 552                    candidate_w_idx_a1[ell, i, b_c[0], b_c[1]]
 553                    for b_c in [(0,0), (0,1), (1,0), (1,1)] if candidate_w_idx_a1[ell, i, b_c[0], b_c[1]] != -1
 554                }
 555                # Compute h[ell,i,a,b,c] for each a,b,c
 556                for a in range(3):
 557                    for b in range(2):
 558                        for c in range(2):
 559                            is_max = ((a == 0) and (b==1) and (ell not in M))
 560                            div = 1e10 if is_max and (i != N-1) else eta[ell]*divergence(mu[ell],z)
 561
 562                            if a == 0:
 563                                h[ell,i,a,b,c] = div + fast_minimization(hminus[children,i,:,:], b -1*(ell not in M), c)[0]
 564
 565                            elif a == 1:
 566                                # G_base: shape (num_children, 2, 2) containing hstar for each child
 567                                G_base = np.array([ hstar[w, i, :, :] for w in children ])
 568                                best_val = 1e20
 569                                best_bs, best_cs, best_w_idx = None, None, -1
 570                                
 571                                # Loop over the small candidate set W_ell(z) (at most 4 children)
 572                                for idx_w in candidate_indices:
 573                                    w = children[idx_w]
 574                                    G = G_base.copy()
 575                                    G[idx_w, :, :] = hplusequal[w, i, :, :]
 576                                    val, bs_cand, cs_cand = fast_minimization(G, b, c - 1*(ell in M))
 577                                    
 578                                    if val < best_val:
 579                                        best_val, best_bs, best_cs, best_w_idx = val, bs_cand, cs_cand, idx_w
 580                                
 581                                h[ell,i,1,b,c] = div + best_val
 582                                # Store pointers
 583                                if best_bs is not None:
 584                                    best_bs_a1[ell,i,b,c,:num_children] = np.array(best_bs[:num_children], dtype=int)
 585                                    best_cs_a1[ell,i,b,c,:num_children] = np.array(best_cs[:num_children], dtype=int)
 586                                    best_w_idx_a1[ell,i,b,c] = int(best_w_idx)
 587
 588                            elif a == 2:
 589                                h[ell,i,a,b,c] = div + fast_minimization(hstar[children,i,:,:], b, c-1*(ell in M))[0]
 590
 591        # Compute the aggregated tables and store pointers for backtracking
 592        for b in range(2):
 593            for c in range(2):
 594                # Compute hplus and store pointer
 595                # Base case: i = N-1
 596                hplus[ell, N-1, b, c] = 1e10
 597                ptr_hplus[ell, N-1, b, c] = -1
 598                
 599                # Iterate downwards from i = N-2 to 0
 600                for i in range(N-2, -1, -1):
 601                    val_from_ip1 = min(h[ell, i+1, 0, b, c], h[ell, i+1, 1, b, c])
 602                    if val_from_ip1 <= hplus[ell, i+1, b, c]:
 603                        hplus[ell, i, b, c] = val_from_ip1
 604                        ptr_hplus[ell, i, b, c] = i + 1
 605                    else:
 606                        hplus[ell, i, b, c] = hplus[ell, i+1, b, c]
 607                        ptr_hplus[ell, i, b, c] = ptr_hplus[ell, i+1, b, c]
 608
 609                # Compute hminus and store pointer
 610                # Base case: i = 0
 611                hminus[ell, 0, b, c] = 1e10
 612                ptr_hminus[ell, 0, b, c] = -1
 613                
 614                # Iterate upwards from i = 1 to N-1
 615                for i in range(1, N):
 616                    val_from_im1 = h[ell, i-1, 2, b, c]
 617                    if val_from_im1 <= hminus[ell, i-1, b, c]:
 618                        hminus[ell, i, b, c] = val_from_im1
 619                        ptr_hminus[ell, i, b, c] = i - 1
 620                    else:
 621                        hminus[ell, i, b, c] = hminus[ell, i-1, b, c]
 622                        ptr_hminus[ell, i, b, c] = ptr_hminus[ell, i-1, b, c]
 623
 624                # Compute hequal, hstar, hplusequal
 625                for i in range(N):
 626                    hequal[ell,i,b,c] = h[ell,i,2,b,c]
 627                for i in range(N):
 628                    hstar[ell,i,b,c] = min(hminus[ell,i,b,c], hequal[ell,i,b,c], hplus[ell,i,b,c])
 629                for i in range(N):
 630                    hplusequal[ell,i,b,c] = min(hequal[ell,i,b,c], hplus[ell,i,b,c])
 631
 632    # Backtracking 
 633    istar = [0 for i in range(K)]
 634    astar = [0 for i in range(K)]
 635    bstar = [0 for i in range(K)]
 636    cstar = [0 for i in range(K)]
 637
 638    istar[kstar] = N-1
 639    bstar[kstar] = 1
 640    cstar[kstar] = 1
 641    astar[kstar] = 0
 642    theoretical_vstar2 = h[kstar, N-1, 0, 1, 1]
 643
 644    # Compute the optimal solution using stored pointers where needed
 645    for ell in list(T.nodes()):
 646        children = list(T[ell])
 647        if children:
 648            if astar[ell] == 0:
 649                (vs,bs,cs) = fast_minimization(hminus[children,istar[ell],:,:], bstar[ell] -1*(ell not in M), cstar[ell])
 650                for i,v in enumerate(children):
 651                    bstar[v] = bs[i]
 652                    cstar[v] = cs[i]
 653                    istar[v] = np.argmin(h[ v,:,2,bstar[v],cstar[v]]  +  (1e10)*(grid >=  grid[istar[ell]]))
 654
 655            elif astar[ell] == 1:
 656                # Read stored bs/cs and chosen special child index
 657                num_children = len(children)
 658                bs = best_bs_a1[ell, istar[ell], bstar[ell], cstar[ell], :num_children].astype(int)
 659                cs = best_cs_a1[ell, istar[ell], bstar[ell], cstar[ell], :num_children].astype(int)
 660                wi = int(best_w_idx_a1[ell, istar[ell], bstar[ell], cstar[ell]])
 661    
 662                for i,v in enumerate(children):
 663                    bstar[v] = int(bs[i])
 664                    cstar[v] = int(cs[i])
 665                    
 666                    i_parent = istar[ell]
 667                    b_child = bstar[v]
 668                    c_child = cstar[v]
 669
 670                    if i == wi:
 671                        # Special child logic, use hplusequal
 672                        val_plusequal = hplusequal[v, i_parent, b_child, c_child]
 673                        if hplus[v, i_parent, b_child, c_child] == val_plusequal:
 674                            istar[v] = ptr_hplus[v, i_parent, b_child, c_child]
 675                        else: # hequal must be the minimum
 676                            istar[v] = i_parent
 677                    else:
 678                        # Other children logic, use hstar
 679                        val_star = hstar[v, i_parent, b_child, c_child]
 680
 681                        # Check branches in order. If values are identical, this establishes a preference.
 682                        # Preference: z_v > z_ell (hplus) -> z_v = z_ell (hequal) -> z_v < z_ell (hminus)
 683                        if hplus[v, i_parent, b_child, c_child] == val_star:
 684                            istar[v] = ptr_hplus[v, i_parent, b_child, c_child]
 685                        elif hequal[v, i_parent, b_child, c_child] == val_star:
 686                            istar[v] = i_parent
 687                        else: # hminus must be the minimum
 688                            istar[v] = ptr_hminus[v, i_parent, b_child, c_child]
 689
 690            elif astar[ell] == 2:
 691                (vs,bs,cs) = fast_minimization(hstar[children,istar[ell],:,:], bstar[ell], cstar[ell]-1*(ell in M))
 692                for i,v in enumerate(children):
 693                    bstar[v] = bs[i]
 694                    cstar[v] = cs[i]
 695
 696                    i_parent = istar[ell]
 697                    b_child = bstar[v]
 698                    c_child = cstar[v]
 699                    
 700                    # Same logic as the other children in a==1
 701                    val_star = hstar[v, i_parent, b_child, c_child]
 702                    
 703                    if hplus[v, i_parent, b_child, c_child] == val_star:
 704                        istar[v] = ptr_hplus[v, i_parent, b_child, c_child]
 705                    elif hequal[v, i_parent, b_child, c_child] == val_star:
 706                        istar[v] = i_parent
 707                    else: # hminus must be the minimum
 708                        istar[v] = ptr_hminus[v, i_parent, b_child, c_child]
 709
 710            for i,v in enumerate(children):
 711                if istar[v] <= istar[ell]:
 712                    astar[v] = 2
 713                elif h[v,istar[v],1,bstar[v],cstar[v]] <  h[v,istar[v],0,bstar[v],cstar[v]]:
 714                    astar[v] = 1
 715                else:
 716                    astar[v] = 0
 717
 718    lambdastar2 = np.array([grid[istar[i]] for i in range(K)])
 719    vstar2 = np.sum(eta*divergence(mu,lambdastar2))
 720
 721    # pick the best of cases 1 and 2
 722    if vstar1 < vstar2:
 723        vstar = vstar1
 724        lambdastar = lambdastar1
 725    else:
 726        vstar = vstar2
 727        lambdastar = lambdastar2
 728
 729    return (lambdastar,vstar)

 733def subgradient_descent(G,mu,N,I,nb_modes,timed=False):
 734    """Computes the optimal solution of the Graves-Lai optimization problem using subgradient descent. Uses values of penalization and step size suggested by the theoretical analysis. Alternative to slsqp.
 735    Args:
 736        G (networkx graph): The graph G
 737        mu (numpy array): The values of function mu
 738        N (int): The number of grid points used for discretization
 739        I (int): The number of iterations of subgradient descent
 740        nb_modes (int): The number of modes of lambda 
 741
 742    Kwargs:
 743        timed (boolean): Set to True to compute the runtime
 744
 745    Returns:
 746        (eta_final,value,runtime) (numpy array,float,float): The optimal solution, the corresponding value of P_GL and the runtime
 747
 748    """
 749    if timed:
 750        start_time=time.time()
 751    kstar = np.argmax(mu)
 752    Delta = mu[kstar] - mu    
 753    K=mu.shape[0]
 754    eta = np.zeros(K)
 755    gamma=0
 756    non_zero_gaps = Delta[np.nonzero(Delta)]
 757    if non_zero_gaps.size == 0:
 758        return (np.ones(K) / K, 0.0)
 759    for k in range(K):
 760        if (Delta[k] > 0):
 761            eta[k] = 1/divergence(mu[k],mu[kstar])
 762            if 2*Delta[k]*eta[k] > gamma:
 763                gamma=2*Delta[k]*eta[k]               
 764    B=eta.dot(Delta)/np.min(non_zero_gaps)
 765    C=np.linalg.norm(Delta)+gamma*K**(3/2)*(mu[kstar]-np.min(mu))**2 #for gaussian distributions we can take A(mu)=mu^*-mu_*
 766    eta_mean = eta/I
 767    delta=np.sqrt(K*B**2/(I*C**2))
 768    for i in range(I-1):
 769        (lambdastar,vstar) = regression_all(G,mu,eta,N,nb_modes)
 770        subgradient = Delta - gamma*divergence(mu,lambdastar)*(sum(eta*divergence(mu,lambdastar)) < 1)  
 771        eta = eta - delta*subgradient
 772        eta[eta < 0] = 0
 773        eta_mean += eta/I
 774    eta_mean[kstar] = 0
 775    (lambdastar,vstar) = regression_all(G,mu,eta_mean,N,nb_modes)
 776    eta_final=eta_mean/sum(eta_mean*divergence(mu,lambdastar))
 777    value=sum(eta_final*Delta)
 778    
 779    if timed:
 780        end_time = time.time()
 781        runtime = end_time - start_time
 782        return(eta_final,value,runtime)
 783    else:
 784        return(eta_final,value)

 786def slsqp(G,mu,N,I,nb_modes,local=False,eta0_guess=None):
 787    """Computes the optimal solution of the Graves-Lai optimization problem using the Python sequential least squares programming implementation (SLSQP). Alternative to subgradient_descent.
 788
 789    Args:
 790        G (networkx graph): The graph G
 791        mu (numpy array): The values of function mu
 792        N (int): The number of grid points used for discretization
 793        I (int): The number of iterations
 794        nb_modes (int): The number of modes of lambda 
 795
 796    Kwargs:
 797        local (boolean): Add a local search constraint if True 
 798        eta0_guess (numpy array): Initial guess for eta, if any 
 799
 800    Returns:
 801        sol (OptimizeResult object): The optimal solution
 802
 803    """
 804
 805    # If local=True, computes optimal rates under a local search constraint
 806    kstar = np.argmax(mu)
 807    Delta = mu[kstar] - mu
 808    K = mu.shape[0]
 809    if eta0_guess is not None:
 810       eta0 = eta0_guess
 811    else:
 812       eta0 = np.zeros(K)
 813       neighborhood = modes_neighborhood(G,mu)
 814       for k in range(K):
 815           if (Delta[k] > 0):
 816               eta0[k] = 1/divergence(mu[k],mu[kstar]) # Lai-Robbins rates as initial guess
 817    if local: # Compute optimal solution of P_GL under added constraint eta_k=0 if k is not in the modes neighborhood
 818        lb = [0] * K
 819        ub = [np.inf] * K
 820        neighborhood = modes_neighborhood(G,mu)
 821        for k in range(K):
 822           if k not in neighborhood:
 823               ub[k] = 0.0
 824        bounds=Bounds(lb,ub)
 825    else:
 826        bounds=Bounds([0]*K,[np.inf]*K)
 827
 828    cons = ({'type': 'ineq', 'fun': lambda eta:regression_all(G,mu,eta,N,nb_modes)[1] -1,
 829             'jac' : lambda eta:divergence(mu,regression_all(G,mu,eta,N,nb_modes)[0])})
 830    objective = lambda eta: eta @ Delta
 831    sol=minimize(objective, eta0, jac=lambda eta:Delta, constraints=cons, options={'maxiter':I}, bounds=bounds)
 832    return(sol)

 837class MultimodalOSSB:
 838    """ OSSB algorithm, with various exploration rates.
 839    
 840     Args:
 841           G (networkx graph): The graph G
 842           K (int): The number of arms
 843           T (int): The time horizon
 844           m (int): The number of allowed modes
 845           true_means (numpy array): The values of function mu
 846           
 847     Kwargs:
 848           N (int): The number of grid points used for discretization
 849           I (int): The number of iterations of subgradient descent or slsqp
 850           strategy (string): 'multimodal' (with subgradient descent), 'multimodal slsqp' (with SLSQP), 'local slsqp' (multimodal slsqp with added local search constraint), 'local' (naive local search rates) or 'classical' (rates given by the Lai-Robbins bound)
 851           use_doubling_schedule (boolean): Computes the solution of P_GL every 2^k iterations if True, and every iteration if False
 852     """
 853    def __init__(self, G, K, T, m, true_means, N=100, I=100, strategy="multimodal", use_doubling_schedule=False):
 854        self.G = G
 855        self.K = K
 856        self.T = T
 857        self.m = m
 858        self.N = N
 859        self.I = I
 860        self.strategy = strategy
 861        
 862        # True means and optimal mean (for regret calculation)
 863        self.true_means = np.asarray(true_means)
 864        self.optimal_mean = np.max(self.true_means)
 865        self.cumulative_regret = 0.0
 866        self.regret_history = []  # Stores regret at each timestep
 867        
 868        # Track empirical means and pull counts
 869        self.mu_hat = np.zeros(K)
 870        self.N_pulls = np.zeros(K, dtype=int)
 871        self.eta_hat = None
 872        self.use_doubling_schedule = use_doubling_schedule
 873        if self.use_doubling_schedule:
 874            self.next_update_t = 1 # tracks at each round if we should update the P_GL solution
 875        
 876    def classical_eta(self):
 877        # Classical Graves-Lai exploration rates (1 / divergence)
 878        kstar = np.argmax(self.mu_hat)
 879        Delta = self.mu_hat[kstar] - self.mu_hat
 880        eta = np.zeros(self.K)
 881        for k in range(self.K):
 882            if k != kstar and Delta[k] > 0:
 883                eta[k] = 1 / (divergence(self.mu_hat[k], self.mu_hat[kstar]))
 884        return eta
 885    
 886    def local_eta(self):
 887        # local search rates (1 / divergence in the neighborhood of modes), not a feasible rate for P_GL
 888        kstar = np.argmax(self.mu_hat)
 889        Delta = self.mu_hat[kstar] - self.mu_hat
 890        eta = np.zeros(self.K)
 891        neighborhood=modes_neighborhood(self.G,self.mu_hat)
 892        for k in range(self.K):
 893            if k in neighborhood and k != kstar and Delta[k] > 0:
 894                eta[k] = 1 / (divergence(self.mu_hat[k], self.mu_hat[kstar]))
 895        return eta
 896
 897    def select_arm(self, t):
 898        # Select arm using the desired strategy
 899        if self.strategy == "classical":
 900            eta = self.classical_eta()
 901        elif self.strategy == "local":
 902            eta = self.local_eta()
 903        elif self.strategy == "multimodal":
 904            eta, _ = subgradient_descent(
 905                G=self.G,
 906                mu=self.mu_hat,
 907                N=self.N,
 908                I=self.I,
 909                nb_modes=self.m
 910            )
 911            
 912        elif self.strategy in ["multimodal slsqp", "local slsqp"]: # local slsqp attempts to solve P_GL with the added constraint eta_k=0 when k is not in the modes neighborhood
 913            should_update_eta = False
 914            if not self.use_doubling_schedule:
 915                should_update_eta = True
 916            elif (t + 1) == self.next_update_t:
 917                should_update_eta = True
 918
 919            if should_update_eta:
 920                
 921                is_local_run = (self.strategy == "local slsqp")
 922                
 923                solution = slsqp(
 924                    G=self.G,
 925                    mu=self.mu_hat,
 926                    N=self.N,
 927                    I=self.I,
 928                    nb_modes=self.m,
 929                    local=is_local_run, 
 930                    eta0_guess=None # Can also do a warm start: replace None by self.eta_hat
 931                )
 932                self.eta_hat = solution.x
 933                
 934                if self.use_doubling_schedule:
 935                    self.next_update_t *= 2
 936            
 937            eta = self.eta_hat
 938        else:
 939            raise ValueError(f"Unknown strategy: {self.strategy}")
 940            
 941        exploration_done = all(
 942            self.N_pulls[k] >= eta[k] * np.log(t + 1)
 943            for k in range(self.K)
 944        )
 945        
 946        return (np.argmax(self.mu_hat) if exploration_done 
 947                else np.argmin(self.N_pulls/(eta + 1e-10)))
 948
 949    def update(self, arm, reward):
 950        # Update statistics after arm pull
 951        n = self.N_pulls[arm]
 952        self.N_pulls[arm] += 1    
 953        self.mu_hat[arm] = (n*self.mu_hat[arm]+reward)/(n+1)
 954        instant_regret = self.optimal_mean - self.true_means[arm]
 955        self.cumulative_regret += instant_regret
 956        self.regret_history.append(self.cumulative_regret)
 957        
 958    def get_regret(self):
 959        # Return cumulative regret up to the current timestep
 960        return self.cumulative_regret

1337def run_single_trial(trial_seed, true_means, graph, m, K, T, use_doubling_schedule=False):
1338    """ Runs a single trial of the regret experiment for multimodal OSSB and classical OSSB
1339
1340    Args:
1341        trial_seed (int): The random seed for this trial to ensure reproducibility
1342        true_means (numpy array): The true mean reward for each arm
1343        graph (networkx graph): The graph structure over the arms
1344        m (int): The number of allowed modes
1345        K (int): The number of arms
1346        T (int): The time horizon for the trial
1347
1348    Kwargs:
1349        use_doubling_schedule (bool): If True, enables the doubling schedule for multimodal OSSB
1350
1351    Returns:
1352        (regret_mmslsqp, regret_local, regret_classical) (list,list): Contains the cumulative regret history (a list of floats) for
1353        the 'multimodal slsqp', 'local', and 'classical' strategies, respectively
1354    """
1355    np.random.seed(trial_seed)
1356
1357    def run_one_strategy(strategy):
1358        # Seed the random number generator for this specific run
1359        schedule_for_this_run = use_doubling_schedule if (strategy == "multimodal slsqp" or strategy == "local slsqp")  else False
1360        
1361        bandit = MultimodalOSSB(
1362            G=graph, K=K, T=T, true_means=true_means, m=m, strategy=strategy, use_doubling_schedule=schedule_for_this_run
1363        )
1364        for t in range(T):
1365            arm = bandit.select_arm(t)
1366            reward = np.random.normal(bandit.true_means[arm], 1)
1367            bandit.update(arm, reward)
1368        return bandit.regret_history
1369
1370    # Run each strategy
1371    regret_mmslsqp = run_one_strategy("multimodal slsqp")
1372    regret_local= run_one_strategy("local")
1373    regret_classical = run_one_strategy("classical")
1374
1375    
1376    return regret_mmslsqp, regret_local, regret_classical

174
175    Args:
176        G (networkx graph): The graph G
177        mu (numpy array): The values of function mu
178        eta (numpy array): The values of weights eta
179        p (list): The location of the modes of lambda
180        k (int): The maximal entry of lambda
181        N (int): The number of grid points used for discretization
182
183    Returns:
184        lambdastar (numpy array): The values of minimizer lambda
185
186    """
187    INF = 1e10
188    K = mu.shape[0]
189    kstar = int(np.argmax(mu))
190    grid = np.linspace(np.min(mu), np.max(mu), N)
191    T = nx.bfs_tree(G, k)
192
193    # Allocate arrays
194    f = np.zeros([K, N, 2])              # f[...,0] is u=-1 (<=), f[...,1] is u=+1 (>)
195    fstar = np.zeros([K, N, 2])         # fstar[...,0] = min_{j<=i} f[...,0], fstar[...,1] = min_{j>i} f[...,1]
196    fsquare = np.zeros([K, N])          # f^\diamond = min(fstar[...,0], fstar[...,1])
197    v = np.full([K, N], -1, dtype=int)  # which child attains min_{child} g_child(z)
198    lambdastar = np.zeros(K, dtype=float)
199
200    # Pointers for backtracking 
201    ptr_minus = np.full([K, N], -1, dtype=int)  # argmin index in [0..i] for f[...,0]
202    ptr_plus  = np.full([K, N], -1, dtype=int)  # argmin index in [i+1..N-1] for f[...,1], -1 if none
203
204    # Forward DP: compute f, then fstar/ptr arrays and fsquare
205    for ell in reversed(list(T.nodes())):
206        children = sorted(list(T.successors(ell)))  # deterministic order, just in case
207        # effective eta: make k^\star have infinite weight so its lambda must equal mu^\star
208        e = INF if (ell == kstar) else float(eta[ell])
209
210        # compute f[ell,i,u] for all grid points
211        for i, z in enumerate(grid):
212            div = e * divergence(mu[ell], z)
213            if ell in p:  # ell is allowed to be a mode
214                base = div
215                # sum over children of their fsquare at same z
216                s = 0.0
217                for j in children:
218                    s += fsquare[j, i]
219                f[ell, i, 0] = base + s
220                f[ell, i, 1] = base + s
221            else:
222                # ell cannot be a mode, and needs to have a neighbour whose reward is higher or equal
223                s = 0.0
224                for j in children:
225                    s += fsquare[j, i]
226                f[ell, i, 0] = div + s
227                # compute the extra min_v term for u=+1 if children exist
228                if children:
229                    # for each child compute g_child = min(fstar_child(+1), f_child(-1)) - fsquare_child
230                    min_g_cost = INF
231                    winner = -1
232                    for child in children:
233                        #  f[child, i, 0] is f_child(z,-1)
234                        cand_plus = fstar[child, i, 1]   # min_{w>z} f_child(w,+1)
235                        cand_minus = f[child, i, 0]      # f_child(z, -1)
236                        g_child = min(cand_plus, cand_minus) - fsquare[child, i]
237                        if g_child < min_g_cost:
238                            min_g_cost = g_child
239                            winner = child
240                    v[ell, i] = int(winner)  # store the winning child id
241                    f[ell, i, 1] = div + s + min_g_cost

244
245        # Compute fstar and pointers for ell:
246        # fstar[...,0] = min_{j<=i} f[...,0]  and ptr_minus is the argmin in [0..i]
247        fstar[ell, 0, 0] = f[ell, 0, 0]
248        ptr_minus[ell, 0] = 0
249        for i in range(1, N):
250            if f[ell, i, 0] <= fstar[ell, i-1, 0]:
251                fstar[ell, i, 0] = f[ell, i, 0]
252                ptr_minus[ell, i] = i
253            else:
254                fstar[ell, i, 0] = fstar[ell, i-1, 0]
255                ptr_minus[ell, i] = ptr_minus[ell, i-1]
256
257        # fstar[...,1] = min_{j>i} f[...,1] ; for i = N-1 it is INF and ptr_plus = -1
258        fstar[ell, N-1, 1] = INF
259        ptr_plus[ell, N-1] = -1
260        # iterate backwards to compute min over j > i
261        for i in range(N - 2, -1, -1):
262            # compare immediate next f at i+1 vs min at i+1
263            if f[ell, i+1, 1] <= fstar[ell, i+1, 1]:
264                fstar[ell, i, 1] = f[ell, i+1, 1]
265                ptr_plus[ell, i] = i + 1
266            else:
267                fstar[ell, i, 1] = fstar[ell, i+1, 1]
268                ptr_plus[ell, i] = ptr_plus[ell, i+1]
269
270        #  compute fsquare
271        for i in range(N):
272            fsquare[ell, i] = min(fstar[ell, i, 0], fstar[ell, i, 1])
273
274    # Backward pass using pointers
275    
276    lambdastar[k] = np.max(mu)
277    # iterate nodes in BFS order from root (k)
278    for ell in list(T.nodes()):
279        if ell == k:
280            continue
281
282        parent = next(T.predecessors(ell))
283        # robust nearest-grid index for parent's lambda (avoid float == just in case) 
284        parent_grid_index = int(np.argmin(np.abs(grid - lambdastar[parent])))
285
286        # Check if ell is constrained child
287        grandparent_list = list(T.predecessors(parent))
288        ell_is_constrained = False
289        if grandparent_list:
290            grandparent = grandparent_list[0]
291            if lambdastar[parent] > lambdastar[grandparent] and parent not in p:
292                constrained_child = v[parent, parent_grid_index]
293                if constrained_child == ell:
294                    ell_is_constrained = True
295
296
297
298        
299        # Decide value of lambda_ell using stored pointers
300        
301        if ell_is_constrained:
302            if fstar[ell, parent_grid_index, 1] <= f[ell, parent_grid_index, 0]:
303                j = ptr_plus[ell, parent_grid_index]  # index j > parent_grid_index minimizing f[...,1]
304                if j == -1:
305                    # no strictly greater grid value -> fallback to equality with parent
306                    lambdastar[ell] = lambdastar[parent]
307                else:
308                    lambdastar[ell] = grid[j]
309            else:
310                lambdastar[ell] = lambdastar[parent]
311        else:
312            # unconstrained case: compare fstar(+1) vs fstar(-1)
313            if fstar[ell, parent_grid_index, 1] <= fstar[ell, parent_grid_index, 0]:
314                j = ptr_plus[ell, parent_grid_index]
315                if j == -1:
316                    lambdastar[ell] = lambdastar[parent]
317                else:
318                    lambdastar[ell] = grid[j]
319            else:
320                j = ptr_minus[ell, parent_grid_index]
321                if j == -1:
322                    lambdastar[ell] = grid[0]
323                else:
324                    lambdastar[ell] = grid[j]
325
326    return lambdastar
